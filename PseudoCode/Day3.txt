CLASS Customer {
    ATTRIBUTES:
        name
        address

    CONSTRUCTOR(name, address)
        SET this.name = name
        SET this.address = address

    METHOD displayDetails()
        PRINT "Customer: ", name, address
}

CLASS Restaurant {
    ATTRIBUTES:
        name
        location

    CONSTRUCTOR(name, location)
        SET this.name = name
        SET this.location = location

    METHOD displayDetails()
        PRINT "Restaurant: ", name, location
}

CLASS DeliveryBoy {
    ATTRIBUTES:
        name
        phone

    CONSTRUCTOR(name, phone)
        SET this.name = name
        SET this.phone = phone

    METHOD displayDetails()
        PRINT "Delivery Boy: ", name, phone
}

CLASS Order {
    ATTRIBUTES:
        orderId
        customer      
        restaurant    
        deliveryBoy  

    CONSTRUCTOR(orderId, customerName, customerAddress,
                restaurantName, restaurantLocation,
                deliveryBoyName, deliveryBoyPhone)

        SET this.orderId = orderId
        CREATE new Customer(customerName, customerAddress)
        CREATE new Restaurant(restaurantName, restaurantLocation)
        CREATE new DeliveryBoy(deliveryBoyName, deliveryBoyPhone)

    METHOD displayOrderDetails()
        PRINT "Order ID: ", orderId
        CALL customer.displayDetails()
        CALL restaurant.displayDetails()
        CALL deliveryBoy.displayDetails()
}

MAIN METHOD {
    CREATE order OBJECT OF Order(101,
                                 "Ayushi", "Delhi",
                                 "Foodie's Hub", "Noida",
                                 "Vipin", "9876543210")

    PRINT "--- Order Details ---"
    CALL order.displayOrderDetails()
}

CLASS Patient {
    ATTRIBUTES:
        id
        name
        age
        disease

    CONSTRUCTOR(id, name, age, disease)
        SET this.id = id
        SET this.name = name
        SET this.age = age
        SET this.disease = disease

    METHOD displayDetails()
        PRINT "ID: ", id, " Name: ", name, " Age: ", age, " Disease: ", disease
}

MAIN METHOD {
    CREATE HashMap<Integer, Patient> patientsMap

    // Add patients
    CREATE patient1 = new Patient(101, "Ayushi", 21, "Flu")
    CREATE patient2 = new Patient(102, "Vipin", 30, "Fever")
    ADD patient1 TO patientsMap WITH KEY = 101
    ADD patient2 TO patientsMap WITH KEY = 102

    PRINT "--- Retrieve a Patient ---"
    RETRIEVE patient FROM patientsMap USING KEY 101
    CALL patient.displayDetails()

    PRINT "--- Update a Patient ---"
    RETRIEVE patient FROM patientsMap USING KEY 102
    SET patient.disease = "Migraine"   // update record
    REPLACE in patientsMap with updated patient

    PRINT "--- Delete a Patient ---"
    REMOVE patient FROM patientsMap USING KEY 101

    PRINT "--- Display All Patients ---"
    FOR EACH entry IN patientsMap
        CALL entry.value.displayDetails()
}
